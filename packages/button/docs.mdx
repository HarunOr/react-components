### Sticker Sheet

<div>
  {(() => {
    const variants = ['primary', 'secondary', 'tertiary', 'tertiary-neutral']
    const title = 'My Button Title'
    const brands = [
      'hashicorp',
      'neutral',
      'terraform',
      'consul',
      'nomad',
      'packer',
    ]
    const bgThemeDict = {
      white: 'light',
      'brand-l3': 'brand-l3',
      black: 'dark',
      'gray-1': 'dark',
      brand: 'brand',
    }
    const bgKnob = 'white'
    const background = bgThemeDict[bgKnob]
    const disabled = false
    // [null, 'anchor', 'inbound', 'outbound', 'download']
    const linkType = null
    const icon = {
      position: 'right', // ['left', 'right']
      isAnimated: false,
    }
    const size = 'medium' // ['small', 'medium']
    return brands.map((brand) => {
      const brandKey =
        brand === 'hashicorp' ? 'brand' : brand === 'neutral' ? 'vault' : brand
      const bgToken =
        bgKnob === 'brand-l3'
          ? `--${brandKey}-l3`
          : bgKnob === 'brand'
          ? `--${brandKey}`
          : `--${bgKnob}`
      return (
        <div
          style={{ padding: '24px 0 32px 0', background: `var(${bgToken})` }}
        >
          <div
            style={{
              marginLeft: '16px',
              color:
                background === 'light' || background === 'brand-l3'
                  ? 'var(--gray-3)'
                  : 'var(--white)',
            }}
            className="g-type-label"
          >
            {brand}
          </div>
          <div style={{ marginTop: '8px' }}>
            {variants.map((variant) => {
              return (
                <span key={variant} style={{ marginLeft: '16px' }}>
                  <Button
                    icon={icon}
                    linkType={linkType}
                    disabled={disabled}
                    title={title}
                    size={size}
                    url={'http://www.example.com'}
                    theme={{
                      brand,
                      variant,
                      background,
                    }}
                  />
                </span>
              )
            })}
          </div>
        </div>
      )
    })
  })()}
</div>
